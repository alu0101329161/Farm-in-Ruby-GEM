def initialize(identificacion = nil, nombre = "", tipo = nil, descripcion = "")

  if(tipo == :avicola || tipo == :ganadera)
    @tipo = tipo
  else
    return nil
  end
  @descripcion = descripcion
  @identificacion = identificacion
  @nombre = nombre
end
